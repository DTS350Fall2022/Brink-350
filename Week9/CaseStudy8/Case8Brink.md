```r
timedat <- SDTZ %>%
  mutate(hourofday = hour(Time)) %>%
  mutate(dayofweek = wday(Time, label = TRUE, abbr = FALSE)) %>%
  mutate(weekofyear = week(Time)) %>%
  mutate(monthofyear = month(Time))
```


```r
ophours <- timedat %>%
```
